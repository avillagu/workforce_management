```typescript
calcularTiempoEnEstado(horaInicio: string): string {
  const inicio = DateTime.fromISO(horaInicio);
  const ahora = DateTime.utc();
  const diff = ahora.diff(inicio, ['hours', 'minutes']);
  return `${diff.hours}h ${diff.minutes}m`;
}
```

**Alerta de descanso >30 min:**
```typescript
esAlertaDescanso(estado: string, horaInicio: string): boolean {
  if (estado !== 'descanso') return false;
  const inicio = DateTime.fromISO(horaInicio);
  const ahora = DateTime.utc();
  const minutos = ahora.diff(inicio, 'minutes').minutes;
  return minutos > 30;
}
```

## EJEMPLOS DE IMPLEMENTACIГ“N

### Servicio de Asistencias
```typescript
@Injectable({ providedIn: 'root' })
export class AsistenciasService {
  private readonly API_BASE = environment.apiUrl;

  constructor(private http: HttpClient) {}

  marcarEstado(estado: string): Observable<RegistroAsistencia> {
    return this.http.post<{ data: RegistroAsistencia }>(
      `${this.API_BASE}/asistencias/marcar`,
      { estado }
    ).pipe(map(res => res.data));
  }

  getEstadoActual(grupoId?: string): Observable<EstadoActual[]> {
    const params = grupoId ? { grupo_id: grupoId } : {};
    return this.http.get<{ data: EstadoActual[] }>(
      `${this.API_BASE}/asistencias/estado-actual`,
      { params }
    ).pipe(map(res => res.data));
  }

  getTiempoTotal(usuarioId: string, fecha: string): Observable<TiempoTotal> {
    return this.http.get<{ data: TiempoTotal }>(
      `${this.API_BASE}/asistencias/tiempo-total`,
      { params: { usuario_id: usuarioId, fecha } }
    ).pipe(map(res => res.data));
  }
}
```

### Componente Asistencias (Empleado)
```typescript
@Component({
  selector: 'app-asistencias',
  templateUrl: './asistencias.component.html',
  styleUrls: ['./asistencias.component.scss'],
  changeDetection: ChangeDetectionStrategy.OnPush
})
export class AsistenciasComponent implements OnInit, OnDestroy {
  readonly ESTADOS = [
    { tipo: 'disponible', label: 'Disponible', icon: 'check_circle', color: 'primary' },
    { tipo: 'descanso', label: 'Descanso', icon: 'coffee', color: 'accent' },
    { tipo: 'en_bano', label: 'En baГ±o', icon: 'wc', color: 'warn' },
    { tipo: 'fuera_de_turno', label: 'Fuera de turno', icon: 'logout', color: 'basic' }
  ];

  estadoActual: EstadoTipo | null = null;
  historial: RegistroAsistencia[] = [];
  tiempoTotal: string = '';
  isLoading = false;

  constructor(
    private asistenciasService: AsistenciasService,
    private authService: AuthService,
    private snackBar: MatSnackBar
  ) {}

  ngOnInit(): void {
    this.cargarHistorial();
    this.cargarTiempoTotal();
  }

  marcarEstado(tipo: EstadoTipo): void {
    if (this.isLoading || this.estadoActual === tipo) return;

    this.isLoading = true;
    this.asistenciasService.marcarEstado(tipo).subscribe({
      next: () => {
        this.estadoActual = tipo;
        this.snackBar.open(`Estado "${tipo}" marcado exitosamente`, 'Cerrar', { duration: 3000 });
        this.cargarHistorial();
        this.cargarTiempoTotal();
      },
      error: (error) => {
        this.snackBar.open(error.error?.details || 'Error al marcar estado', 'Cerrar', { duration: 5000 });
      },
      complete: () => {
        this.isLoading = false;
      }
    });
  }

  private cargarHistorial(): void {
    this.asistenciasService.getMisEstados().subscribe({
      next: (data) => {
        this.historial = data;
      }
    });
  }

  private cargarTiempoTotal(): void {
    const usuario = this.authService.currentUser();
    const hoy = DateTime.utc().toFormat('yyyy-MM-dd');
    this.asistenciasService.getTiempoTotal(usuario.id, hoy).subscribe({
      next: (data) => {
        this.tiempoTotal = `${data.horas}h ${data.minutos}m`;
      }
    });
  }
}
```
