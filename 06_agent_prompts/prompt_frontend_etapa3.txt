## EJEMPLOS DE IMPLEMENTACIÓN

### Servicio de Turnos
```typescript
@Injectable({ providedIn: 'root' })
export class TurnosService {
  private readonly API_BASE = environment.apiUrl;

  getTurnos(inicio: string, fin: string): Observable<Turno[]> {
    return this.http.get<Turno[]>(
      `${this.API_BASE}/turnos?fecha_inicio=${inicio}&fecha_fin=${fin}`
    );
  }

  crearTurno(turno: Partial<Turno>): Observable<Turno> {
    return this.http.post<Turno>(`${this.API_BASE}/turnos`, turno);
  }

  programacionMasiva(datos: ProgramacionMasivaDto): Observable<any> {
    return this.http.post(`${this.API_BASE}/turnos/masivo`, datos);
  }
}
```

### Manejo de Superposición
```typescript
if (form.invalid) return;

this.turnosService.crearTurno(form.value).subscribe({
  next: (turno) => {
    this.snackBar.open('Turno programado exitosamente', 'Cerrar', { duration: 3000 });
    this.dialogRef.close(turno);
  },
  error: (error) => {
    if (error.status === 409) {
      this.snackBar.open(
        'Conflicto: El empleado ya tiene un turno en ese horario',
        'Cerrar',
        { duration: 5000 }
      );
    } else {
      this.snackBar.open('Error al programar turno', 'Cerrar', { duration: 3000 });
    }
  }
});
```
